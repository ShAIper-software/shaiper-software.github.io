    A basic comparator for 1D numpy.ndarray objects A and B.¶

    def __eq__(
    self,
    other: "ndarray_comparator",
) -> bool:
    """Implements the equal (==) operator.

    Args:
        other (ndarray_comparator): Value to compare against

    Returns:
        bool: True when self == other, otherwise False
    """
    if self.n != other.n:
        return False

    for i in range(self.n):
        if np.abs(self.value[i] - other.value[i]) > self.eps:
            return False

    return True

    def __gt__(
    self,
    other: "ndarray_comparator",
) -> bool:
    """Implements the greater than (>) operator

    Args:
        other (ndarray_comparator): Value to compare against

    Returns:
        bool: True when self > other, otherwise False
    """

    if self.n > other.n:
        return True

    if self.n < other.n:
        return False

    for i in range(self.n):
        if self.value[-1 - i] > self.eps + other.value[-1 - i]:
            return True

        if self.value[-1 - i] + self.eps < other.value[-1 - i]:
            return False

    return False

    def f_not_implemented(func) -> Callable:
    """A decorator to be used on functions which are not implemented yet. Useful
    when the codebase is large and you are unsure what functions are called and
    where. Aborts the program on all MPI processes and produces a call trace for
    debugging.

    Args:
        func (Callable): A function to be decorated

    Returns:
        Callable: the wrapped function
    """

    @functools.wraps(func)
    def wrapper():
        history = []
        frame = inspect.currentframe().f_back
        while frame is not None:
            if (
                "/logging.py" not in frame.f_code.co_filename
                and "/mpi.py" not in frame.f_code.co_filename
            ):
                history.append((frame.f_code.co_filename, frame.f_lineno))
            frame = frame.f_back
        Logger.warning(
            f"Function: {func.__name__} is not implemented yet, call history:"
        )
        for history_entry in history:
            Logger.warning(f"{history_entry}")

        MPI.COMM_WORLD.Abort(0)

    return wrapper

    def get_index_eq_min(
    u: ndarray_comparator,
    V: np.ndarray,
    V_ordering: list[int],
) -> int:
    """Provided the comparator u and the associated array, find and retrieve the
    highest row index i such that V[i] > u.

    Args:
        u (ndarray_comparator): referential row to compare against
        V (np.ndarray): row space in which to look for the rows
        V_ordering (list[int]): An ordering of rows in V such that\
        V[V_ordering[i]] <= V[V_ordering[i + 1]]

    Returns:
        int: the highest row index i such that V[i] > u
    """

    v_idx_start = 0
    v_idx_end = len(V_ordering) - 1
    while v_idx_end - v_idx_start > 0:
        v_idx_mid = (v_idx_end + v_idx_start) // 2
        # Logger.warning(f'get_index_eq_min: {v_idx_start} - {v_idx_end}')

        Vpointer_ordered = V_ordering[v_idx_mid]
        v = ndarray_comparator(V[Vpointer_ordered], u.eps)
        if v < u:
            v_idx_start = v_idx_mid + 1
        else:
            v_idx_end = v_idx_mid

    return v_idx_start

    def get_maximal_matching(
    graph_edges: set[tuple[int, int]],
) -> set[tuple[int, int]]:
    """A greedy algorithm finding a maximal matching from the provided edges.

    Args:
        graph_edges (set[tuple[int, int]]): Each entry is a tuple (u, v),\
        meaning that vertex u is connected to vertex v

    Returns:
        set[tuple[int, int]]: Subset of the input graph_edges such that adding\
        any edge from: graph_edges - output will connect to an edge already in\
        the output.
    """

    output = set()
    vertices_in_output = set()

    for edge in graph_edges:
        u = edge[0]
        v = edge[1]

        if u not in vertices_in_output and v not in vertices_in_output:
            output.add(edge)
            vertices_in_output.add(u)
            vertices_in_output.add(v)

    return output

    def get_similar_row_pairs(
    U: np.ndarray,
    U_ordering: list[int],
    U_indices: list[int],
    V: np.ndarray,
    V_ordering: list[int],
    V_indices: list[int],
    eps: float,
) -> set[tuple[int, int]]:
    """Takes two arrays U with shape (nU, k) and V with shape (nV, k) and finds
    rows of U which are also in V given the provided tolerance eps > 0. Outputs
    the global row indices of corresponding pairs.

    Args:
        U (np.ndarray): First array containing the first set of rows.
        V (np.ndarray): Second array containing the second set of rows.
        U_ordering (list[int]): An ordering of rows in U such that\
        U[U_ordering[i]] <= U[U_ordering[i + 1]]
        U_indices (list[int]): The array U is possibly distributed among\
        multiple MPI processes. This list keeps track of the global row\
        indices for array U.
        V_ordering (list[int]): An ordering of rows in V such that\
        V[V_ordering[i]] <= V[V_ordering[i + 1]]
        V_indices (list[int]): The array V is possibly distributed among\
        multiple MPI processes. This list keeps track of the global row\
        indices for array V.
        eps (float): tolerance parameter to be used in the comparator

    Returns:
        set: Outputs a set of corresponding pairs, i.e. each (u, v) in the\
        output means that U[u] == V[v]. Outputs an empty set when no matches\
        are found.
    """

    out = set()

    nV = len(V_indices)

    for Upointer in range(len(U_indices)):
        Upointer_ordered = U_ordering[Upointer]
        Uglobal_idx = U_indices[Upointer_ordered]

        u = ndarray_comparator(U[Upointer_ordered], eps)

        # first determine whether it makes sense to look
        v0 = ndarray_comparator(V[V_ordering[0]], eps)
        v1 = ndarray_comparator(V[V_ordering[-1]], eps)
        has_solution = u <= v1 and u >= v0
        if not has_solution:
            continue

        # look for the lowest element in set V greater than u to speed-up the
        # search
        v_from_idx = get_index_eq_min(u, V, V_ordering)

        for Vpointer in range(v_from_idx, nV):
            Vpointer_ordered = V_ordering[Vpointer]
            Vglobal_idx = V_indices[Vpointer_ordered]

            # when global indices are the same, the rows are trivially equal,
            # we are not interested in such cases
            if Vglobal_idx == Uglobal_idx:
                continue
            if u == ndarray_comparator(V[Vpointer_ordered], eps):
                min_val = min(Uglobal_idx, Vglobal_idx)
                max_val = max(Uglobal_idx, Vglobal_idx)
                out.add((min_val, max_val))
            else:
                break

    return out

    def get_simplex_distance(
    *,
    q: np.ndarray,
    points: np.ndarray,
    simplices: list[tuple[int, ...]],
) -> float:
    """Calculates the shortest distance from point q to any point lying in the
    provided simplex set.

    Args:
        q (np.ndarray): Coordinates of the point to calculate for.
        points (np.ndarray): Coordinates of the simplex vertices
        simplices (list[tuple[int, ...]]): Simplex adjacency

    Returns:
        float: min ||q - x|| over x, where x is any point in the provided\
        simplex set.
    """
    qw_distance = None
    for S in simplices:

        # for each simplex S, find w in S such that ||q-w||=min||q-x||
        # over x from S

        # the algorithm used is a genral purpose least square solution
        # applicable to simplices of any dimension

        u = S[0]
        b = (q - points[u]).reshape(-1, 1)
        A = []
        for i in range(1, len(S)):
            v = S[i]
            A.append((points[v] - points[u]).reshape(-1, 1))

            u = v

        A = np.concatenate(A, axis=1)
        (sol, res, rank, singular_values) = np.linalg.lstsq(A, b, rcond=None)

        sol = sol.flatten()
        for i in range(len(sol)):
            if sol[i] < 0:
                sol[i] = 0

            if sol[i] > 1:
                sol[i] = 1

        u = S[0]
        w = points[u].copy()
        for i in range(len(sol)):
            v = S[i + 1]
            w += sol[i] * (points[v] - points[u])
            u = v

        if qw_distance is None:
            qw_distance = np.linalg.norm(q - w)
        else:
            qw_distance = min(qw_distance, np.linalg.norm(q - w))

    return qw_distance

    def permute_columns(
    data: list[np.ndarray],
    keys_current: list[str],
    keys_new: list[str],
) -> list[np.ndarray]:
    """Given the list of numpy arrays data and its column labeling keys_new,
    find and apply a column permutation such that the data and its column
    labeling corresponds to the labeling keys_current.

    Args:
        data (list[np.ndarray]): List of arrays to permute
        keys_current (list[str]): Target column ordering
        keys_new (list[str]): Current column ordering

    Returns:
        list[np.ndarray]: a list of elemenets in data (in the same order) with\
        permuted columns.
    """
    keys_indexing_new = {}
    incorrect_labels = []
    for i, key_new in enumerate(keys_new):
        keys_indexing_new[key_new] = i

        if key_new not in keys_current:
            incorrect_labels.append(key_new)

    current2new_key_map = []
    for key_current in keys_current:
        if key_current in keys_indexing_new:
            current2new_key_map.append(keys_indexing_new[key_current])

    if len(current2new_key_map) != len(keys_new):
        msg = f"Incorrect column labeling encountered!. Incorrect labels: {incorrect_labels}."
        Logger.warning(msg)
        raise ValueError(msg)

    data_new = [np.array([D[i] for i in current2new_key_map]) for D in data]

    return data_new

    def transform_function(
    func,
) -> Callable:
    """A decorator to be used around functions producing data transformations.
    The decorator checks whether the input contains the required argument and
    whether the output contains the required arguments.

    Checks for missing arguments. In case of a failure aborts the program on
    all MPI processes.
    Checks and informs the user if any error in input/output definition is
    present.

    Args:
        func (Callable): A function to be decorated

    Returns:
        Callable: the wrapped function
    """

    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        if "data" not in kwargs:
            Logger.warning(
                f"Function: {func.__name__} is missing a required\
                argument: data"
            )
            MPI.COMM_WORLD.Abort(0)

        result = func(*args, **kwargs)

        if not isinstance(result, dict):
            Logger.warning(
                f"Function: {func.__name__} has incorrect output type: requires\
                a dictionary"
            )
            MPI.COMM_WORLD.Abort(0)

        if "data" not in result:
            Logger.warning(
                f"Function: {func.__name__} is missing a required output field:\
                data"
            )
            MPI.COMM_WORLD.Abort(0)

        return result

    return wrapper
