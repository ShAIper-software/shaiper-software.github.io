    @classmethod
def validate_actions(
    cls,
    *,
    actions: dict,
    operation_tasks: dict,
):
    """Validates the series of actions in the configuration file.

    Args:
        actions (dict): A dictionary containing action keys, objects and\
            object functions with their parameters to be called in each\
            action.
        operation_tasks (dict): To be filled with action indices.

    Raises:
        ValueError: This is raised when the actions do not satisfy the\
            naming convention 'action_N' with N being an integer.
        AttributeError: This is raised when the specified function does not\
            exist as a member function of the object.
    """
    keys = list(actions.keys())

    indices = []
    for k in keys:
        if not k.startswith("action_"):
            msg = f"Invalid action key: {k}"
            Logger.warning(msg)
            raise ValueError(msg)

        tmp_ = k.split("_")
        if len(tmp_) > 2:
            msg = f"Invalid action key: {k}. Expected: operation_N, where N is an integer >= 0"
            Logger.warning(msg)
            raise ValueError(msg)

        action_index = int(tmp_[-1])

        indices.append(action_index)

        str_len = len(tmp_[-1].strip())
        if action_index == 0:
            N_len = 1
        else:
            N_len = int(np.floor(np.log10(action_index))) + 1

        if N_len < str_len:
            msg = f'Action key in the form "action_N" requires that "N" is the shortest representation of that integer, i.e. 5 is OK, 00005 is not.'
            raise ValueError(msg)

    indices = sorted(indices)

    for action_index in indices:
        k = f"action_{action_index}"

        fname = actions[k].fname
        object_uid = actions[k].object_uid
        parameters = actions[k].parameters

        ncurrent_actions = runtime_state.get_n_actions()
        runtime_state.add_action(
            object_uid=object_uid,
            fname=fname,
            parameters=parameters,
        )
        operation_tasks["actions"].append(ncurrent_actions)

    @classmethod
def validate_object_references(
    cls,
    *,
    actions: dict,
    objects: dict,
):
    """Checks if the actions reference objects defined in the configuration
    hierarchy.

    Args:
        actions (dict): Action sequence to be validated.
        objects (dict): Objects contained in the same configuration file.

    Raises:
        ValueError: This is raised when the object associated with some\
            action is not specified in the same configuration file or in\
            previously parsed configuration files.
    """
    for action_id, action in actions.items():
        if action.object_uid is None:
            continue

        if action.object_uid not in objects:
            if not runtime_state.contains_uid(action.object_uid):
                msg = (
                    f"Action '{action_id}' references unknown object_id"
                    "'{action.object_uid}'"
                )
                Logger.warning(msg)
                raise ValueError(msg)
            else:
                msg = (
                    f'Action "{action_id}" references an object '
                    f'"{action.object_uid}" from a different configuration file.'
                )
                Logger.warning(msg)

    @classmethod
def validate_objects(
    cls,
    *,
    objects: dict,
    operation_tasks: dict,
):
    """Checks the validity of the objects defined in the configuration file.

    Args:
        objects (dict): Contains unique identifiers, class names and\
            parameters for each object to be initialized.
        operation_tasks (dict): Is filled with object configurations.

    Raises:
        ValueError: This is raised when multiple objects with the same\
            identifier are being specified anywhere in the configuration\
            hierarchy. It is also raised when an object has an identifier\
            in the form 'action_N', wehere N is a nonnegative integer.
        AttributeError: This is raised when the object class does not exist.
    """
    pattern = re.compile(r"^action_(\d+)$")

    for k, v in objects.items():
        if runtime_state.contains_uid(k):
            msg = (
                f'An object with ID: "{k}" is already defined.'
                "Check your configuration files."
            )
            Logger.warning(msg)
            # raise ValueError(msg)

        if pattern.match(k):
            msg = (
                f"Objects cannot have unique identifiers in the form "
                '"action_N", where N is a nonnegative integer.'
                f"but encountered an object with UID: {k}."
            )
            Logger.warning(msg)
            raise ValueError(msg)

        constructor = v.constructor
        parameters = v.parameters
        # split into module and object

        runtime_state.add_object_configuration(
            key=k,
            constructor=constructor,
            parameters=parameters,
        )
        operation_tasks["objects"].append(k)

    @model_validator(mode="after")
@classmethod
def validate_operation(cls, field_value):

    operation_tasks = {
        "objects": [],
        "actions": [],
    }

    ObjectDefinition.validate_objects(
        objects=field_value.objects,
        operation_tasks=operation_tasks,
    )

    ActionDefinition.validate_actions(
        actions=field_value.actions,
        operation_tasks=operation_tasks,
    )
    ActionDefinition.validate_object_references(
        actions=field_value.actions,
        objects=field_value.objects,
    )

    runtime_state.add_operation(operation_tasks)

    return field_value
