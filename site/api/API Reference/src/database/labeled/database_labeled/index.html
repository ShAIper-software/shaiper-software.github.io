    @Logger.logged()
def __init__(
    self,
    path_tgt: str,
) -> None:
    """Initializes the database in a target directory. If it exists, loads
    relevant metadata. If it doesnt exist, creates new directory and
    metadata.

    Args:
        path_tgt (str): defines a path to the top directory of the database
    """

    self.path_main = path_tgt
    self.rank, self.world_size = mpi.get_rank(), mpi.get_world_size()

    self.data_loaded = {}
    self.data = {}
    self.local2global_index_map = {}
    self.global2local_index_map = {}

    self.metadata = self.get_metadata()

    for key in self.get_data_keys():
        self.data_loaded[key] = False
        self.data[key] = []

    Logger.info(f'Initializing Database: "{self.path_main}"')

    if mpi.get_rank() == 0:
        os.makedirs(f"{self.path_main}/data", exist_ok=True)
    mpi.sync()
    self.database_file_controller = ParallelFile(f"{self.path_main}/data")

    self,

    """Adds a transformation to a list of transformations performed on the

                "operations_history"
            )
            + [{"function": dill.dumps(f), "kwargs": kwargs_tmp}],
        }
    )

    @Logger.logged()
def append_new_data(
    self,
    data_local: dict[str],
    source_data_code: str,
):
    """Takes the data provided in the data_local dictionary and appends them
    to this database. Adds the data source code to the list of already known
    sources.

    Args:
        data_local (dict[str]): A dictionary containing\
        (prefix -> data) entries. For each prefix, append the data to the\
        end of the database with the corresponding prefix key.
        source_data_code (str): Code of the data source.
    """
    mpi.sync()
    n_local = 0
    for _, v in data_local.items():
        n_local = max(len(v), n_local)
    nentries_to_store_local = n_local
    nentries_to_store_global = mpi.allreduce(
        nentries_to_store_local, mpi.OP.SUM
    )

    self.database_file_controller.append(
        data=data_local,
        msg=f"Appending new data entries for keys ({data_local.keys()})",
    )

    for key in data_local.keys():
        self.add_data_key(key)

    self.add_n_entries(nentries_to_store_global)
    self.add_data_code(source_data_code)
    self.data_changed()

    """Deletes all data entries and metadata.

    """

    self.database_file_controller.reset()

    @staticmethod
@Logger.logged()
def copy_database(
    *,
    database_dir_src: str,
    database_dir_tgt: str,
) -> None:
    """Copies the contents of an existing Database into this Database. This
    Database will be overwritten by this process.

    Should be called by ALL MPI processes.

    Args:
        database_dir_src (str): Path to the directory of the existing\
        database
        database_dir_tgt (str): Path to the directory of the new database
    """

    if mpi.is_master():

        # check if database exists, if yes, copies the content
        if DatabaseLabeled.is_database_directory(database_dir_src):
            if os.path.exists(database_dir_tgt):
                Logger.warning(
                    f"Target directory: '{database_dir_tgt}' already\
                    exists, removing contents"
                )
                shutil.rmtree(database_dir_tgt)

            Logger.info(f"Copying database from '{database_dir_src}'")
            shutil.copytree(database_dir_src, database_dir_tgt)
        else:
            Logger.info(
                f"Path {database_dir_src} does not contain a database,\
                doing nothing."
            )

    mpi.sync()

    """Clears all chaches and supplementary structures. Called after the

        self.data_loaded[key] = False
        self.data[key] = []

    mpi.sync()

@Logger.logged()
def find_databases(dir_tgt: str) -> list[str]:
    """Looks inside the specified directory and recursively determines all
    subdirectories containing a database.

    If a path A/B/C contains a database, then the recursion stops, i.e.
    potential databases in A/B/C/* will not be discovered.

    Args:
        dir_tgt (str): Specifies which directory should be searched for for\
        the databases.

    Returns:
        list[str]: A list of strings representing absolute paths to the\
        databases so they can be used as an argument to a database\
        constructor
    """

    Logger.info(
        f'Looking for databases in directory "{dir_tgt}" and all its\
        subdirectories'
    )
    # check if the path exists
    if not os.path.exists(dir_tgt):
        Logger.info(f'Directory "{dir_tgt}" does not exist')
        return []

    # when it exists, it must be a directory
    if not os.path.isdir(dir_tgt):
        Logger.info(f'"{dir_tgt}" is not a directory')
        return []

    working_dir = os.getcwd()
    # first check if the specified directory itself is a Database
    if DatabaseLabeled.is_database_directory(dir_tgt):
        return [f"{working_dir}/{dir_tgt}"]

    tmp_ = [
        f"{working_dir}/{Path(dir_tgt)}/{d.name}"
        for d in Path(dir_tgt).iterdir()
        if d.is_dir()
    ]
    output = []
    # we iteratively check if the subdirectories contain a database
    # and if not, include subdirectories instead
    while len(tmp_) > 0:
        tmp2_ = []
        for d_ in tmp_:
            if DatabaseLabeled.is_database_directory(d_):
                output.append(d_)
            else:
                tmp2_ += [
                    f"{d_}/{d.name}"
                    for d in Path(d_).iterdir()
                    if d.is_dir()
                ]
        tmp_ = tmp2_

    return output

def load_stats(
    self,
    *,
    stat_code_value: tuple[str, StatisticsType],
    data_keys: list[str],
) -> dict[str:Any]:
    """Loads the supplied database statistic for the specified database
    prefixes.

    Args:
        stat_code_value (tuple[str, StatisticsType]): statistics code name\
        and type.
        data_keys (list[str]): A list of database prefix keys to be loaded.

    Returns:
        dict[str: Any]: Calculated statistics in a (key -> value(s))\
        dictionary.
    """
    if not self.stats_calculated(
        stat_code_value=stat_code_value, data_keys=data_keys
    ):
        msg = f'Statistic "{stat_code_value[0]}" has not been calculated\
        yet and cannot be loaded.'
        Logger.info(msg)
        mpi.abort(5, msg)

    if stat_code_value[1] == StatisticsType.COL:
        output = {}
        if mpi.get_rank() == 0:
            # if the stat file exists, we load it and return the contained
            # value
            for i, key in enumerate(data_keys):
                stats_path = self.get_stats_path(
                    stat_code_value=stat_code_value, key=key
                )
                if os.path.isfile(stats_path):
                    with open(stats_path, "rb") as f:
                        output[key] = pickle.load(f)
                else:
                    for j, key2 in enumerate(data_keys[i:]):
                        k_ = f"{key}-{key2}"
                        stats_path = self.get_stats_path(
                            stat_code_value=stat_code_value, key=k_
                        )
                        if os.path.isfile(stats_path):
                            with open(stats_path, "rb") as f:
                                output[k_] = pickle.load(f)

        return mpi.broadcast(output, root=0)

    elif stat_code_value[1] == StatisticsType.ROW:
        output = {}
        for key in data_keys:
            stat_dir = self.get_directory_stats(
                stat_type=stat_code_value[1], key=key
            )
            stat_file_path = f"{stat_dir}/{stat_code_value[0]}"
            pio_file = ParallelFile(stat_file_path)

            stats, _ = pio_file.load(
                keys=[key],
                msg=f"Loading statistic: '{stat_code_value[2]}' for keys:\
                {[key]}",
            )
            output[key] = stats[key]

        return output

    else:
        mpi.abort(0, f"Uknown Statistics Type: {stat_code_value[1]}")

    return None

    self,
    train_ratio: float,
    test_ratio: float,
    validation_ratio: float,

    """Randomly splits the entries in this database into three disjoint 
    sets for training, testing and validation. Assumes the training ratio is
    greater than zero. When the ratios sum up to values greater than one, 
    clips the values.

    Args:
        train_ratio (float): What portion of this database should be used\
            for training.
        test_ratio (float): What portion of this database should be used\
            for testing.
        validation_ratio (float): What portion of this database should be\
            used for validation.

    Raises:
        ValueError: When the supplied ratios are lower than zero.
    """

    if train_ratio <= 0 or test_ratio < 0 or validation_ratio < 0:
        msg = f"Incorrect database split ratios: {train_ratio}, {test_ratio}, {validation_ratio}"
        Logger.warning(msg)
        raise ValueError(msg)

    Logger.info(
        f"Calculating database split for train ratio: {train_ratio}, test ratio: {test_ratio}, validation ratio: {validation_ratio}"
    )
    nglob = self.get_n_entries_global()

    n_train = int(nglob * train_ratio)
    n_test = int(nglob * test_ratio)
    n_validation = int(nglob * validation_ratio)
    if n_train > nglob:
        n_train = nglob
        n_test = 0
        n_validation = 0

    if n_train + n_test > nglob:
        n_test = nglob - n_train
        n_validation = 0

    if n_train + n_test + n_validation > nglob:
        n_validation = nglob - n_train - n_test

    Logger.info(
        f"Database will be split into {n_train} training, {n_test} testing and {n_validation} validation samples."
    )

    P = np.random.permutation(nglob)
    P = mpi.broadcast(P)
    train_indices = P[:n_train]
    test_indices = P[n_train : n_train + n_test]
    validation_indices = P[
        n_train + n_test : n_train + n_test + n_validation
    ]

    # store the values into the metadata field
    self.metadata_set_values(
        values={
            "split_train_indices": train_indices,
            "split_test_indices": test_indices,
            "split_validation_indices": validation_indices,
        }
    )
