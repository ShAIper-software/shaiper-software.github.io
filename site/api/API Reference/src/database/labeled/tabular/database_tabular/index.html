    """Uses the provided Importers to parse the Tabular data from the
    specified file/directory. The Database is then updated. Applies all
    transformations already included in the history of this database

    Args:
        importers (list[ImporterTabular]): User provided importer instances

            matter.
    """

    # if the data has already been parsed, we ignore this request
    for importer in importers:
        data_code = importer.get_importer_code()
        if data_code is not None:
            if self.contains_data(data_code):
                continue

        (
            data_new,
            labels_new,
        ) = importer.get_data()

        dimensions = self.metadata_get_value("dimensions")
        data_keys = list(data_new.keys())

            if data_key not in dimensions:
                dimensions[data_key] = 0

                dim = len(labels_new[data_key])
                self.metadata_set_values(
                    {
                        "labels": {
                            data_key: labels_new[data_key],
                        },

                            data_key: dim,

        # what if the newly added data is correct, but has permuted columns?
        data = {}
        Logger.warning(f"DATA_KEYS: {labels_current_dict}")
        for data_key in data_keys:
            data[data_key] = permute_columns(
                data_new[data_key],
                labels_current_dict[data_key],
                labels_new[data_key],
            )

        #     for k, v in kwargs.items():

        #     results = f(
        #         data = data,
        #         **kwargs,
        #     )

        #     data = results['data']

        # check for consistency of the newly appended data
        self.append_new_data(
            data_local=data,
            source_data_code=data_code,
        )

    self.remove_duplicate_rows(
        duplicate_row_eps=1e-9,
    )

    @Logger.logged()
def add_new_column(
    self,
    tgt_data_key: str,
    new_column_name: str,
    src_data_columns: list[tuple[str, int]],
    op_func: Callable,
) -> None:
    """Augments the the data of this database by adding a new column as
    a function of existing columns.

    Args:
        tgt_data_key (str): The database prefix key for the augmentation\
        to be added.
        new_column_name (str): The label associated with the newly created\
        column.
        src_data_columns (list[tuple[str, int]]): A list of database prefix\
        and index pairs to be used as inputs to the augmentation function.
        op_func (Callable): The augmentation function.
    """

    current_labels = self.metadata_get_value("labels")
    current_dimensions = self.metadata_get_value("dimensions")
    current_transforms = self.metadata_get_value("transforms")

    keys_to_load = set([tgt_data_key])
    for k, _ in src_data_columns:
        keys_to_load.add(k)

    self.load_database(keys=list(keys_to_load))

    results = self._create_new_column(
        data=self.data,
        tgt_data_key=tgt_data_key,
        src_data_columns=src_data_columns,
        op_func=op_func,
    )

    self.add_transformation(
        f=self._create_new_column,
        kwargs={
            "tgt_data_key": tgt_data_key,
            "new_column_name": new_column_name,
            "src_data_columns": src_data_columns,
            "op_func": op_func,
        },
    )

    current_labels[tgt_data_key].append(new_column_name)
    current_dimensions[tgt_data_key] += 1
    current_transforms[tgt_data_key] = (
        np.append(current_transforms[tgt_data_key][0], 1),
        np.append(current_transforms[tgt_data_key][1], 0),
    )
    self.update_metadata_value(
        key="labels",
        values=current_labels,
    )
    self.update_metadata_value(
        key="dimensions",
        values=current_dimensions,
    )
    self.update_metadata_value(
        key="transforms",
        values=current_transforms,
    )

    self.database_file_controller.update(
        indices_global={
            tgt_data_key: self.local2global_index_map[tgt_data_key]
        },
        data=results["data"],
        msg=f"Updating entries with key: {tgt_data_key} after creating a\
        new column: {new_column_name}",
    )
    self.data_changed()

    @Logger.logged()
def calculate_statistic(
    self,
    *,
    stat_code: StatCode,
    data_keys: list[str],
) -> None:
    """Method calculating the supplied statistic depending on
    the data retrieved from the StatCode Enumerator for each
    of the database prefix keys specified.

    Args:
        stat_code (StatCode): Enumerator containing instructions for
            calculating the statistic.
        data_keys (list[str]): Database prefix keys for which to calculate
            the statistic.
    """
    calculated_stats = self.stats_calculated(
        stat_code_value=stat_code.value,
        data_keys=data_keys,
    )

    # which keys of the supplied ones are missing this statistic?
    keys_to_calculate = []
    for k, v in calculated_stats.items():
        if not v:
            keys_to_calculate.append(k)

    if len(keys_to_calculate) == 0:
        Logger.info(f"{stat_code.value[2]} have been already calculated")
        return

    Logger.info(
        f"'{stat_code.value[2]}' have either not been calculated, or the\
        data has since changed and recalculation is required"
    )

    values_ = None

    # the the Enumerator contains the required data, we can continue.
    if len(stat_code.value) >= 5:

        # some statistics may depend on other statistics, we need to
        # calculate those first
        if "dependencies" in stat_code.value[4].keys():
            dependencies_calculated = {}
            for dependency_key, dependency_code in stat_code.value[4][
                "dependencies"
            ].items():
                dep_value = self.get_statistic(
                    stat_code=dependency_code, data_keys=keys_to_calculate
                )
                dependencies_calculated[dependency_key] = dep_value
            stat_code.value[4][
                "dependencies_calculated"
            ] = dependencies_calculated

        values_ = stat_code.value[3](
            tabular_dataset=self,
            keys=keys_to_calculate,
            **stat_code.value[4],
        )
        self.update_stats(stat_code_value=stat_code.value, data=values_)

    else:
        mpi.abort(
            1,
            f"'{stat_code.value[2]}' calculation not implemented in Tabular\
            Database.",
        )

    @Logger.logged()
def cleanup_columns(
    self,
    *,
    keys_to_consider: list[str],
    correlation_eps: float,
    variance_eps: float,
) -> None:
    """Calls a suite of column cleaning methods. As of now, calls two
    methods, one based on column variance and the other on the column
    correlation.

    Args:
        keys_to_consider (list[str]): Database prefix keys for which the\
        cleaning process is executed.
        correlation_eps (float): Epsilon value for determining when a\
        column is correlated to other columns.
        variance_eps (float): Epsilon value for determining when a column\
        has zero variance or not.
    """

    column_indices_A = self._remove_columns_with_no_variance(
        variance_eps=variance_eps, keys=keys_to_consider
    )
    column_indices_B = self._remove_columns_with_high_correlation(
        correlation_eps=correlation_eps, keys=keys_to_consider
    )

    removed_columns = {
        k: column_indices_A[k] | column_indices_B[k]
        for k in keys_to_consider
    }

    nremoved = np.sum(
        [len(removed_columns[key]) for key in keys_to_consider]
    )
    if nremoved > 0:
        labels = self.metadata_get_value("labels")
        for key in keys_to_consider:
            if len(removed_columns[key]) > 0:
                Logger.info(
                    f"Removed columns [{key}]:\
                    {[(idx, labels[key][idx]) for idx in removed_columns[key]]}"
                )

        dimensions = self.metadata_get_value("dimensions")
        kept_columns = {
            k: np.sort(
                list(
                    set(range(0, dimensions[k])).difference(
                        removed_columns[k]
                    )
                )
            )
            for k in keys_to_consider
        }

        data_tmp, labels_tmp, dimensions_tmp, transforms_tmp = (
            self.prune_columns(
                kept_columns=kept_columns,
            )
        )

        self.update_metadata_value(
            key="labels",
            values=labels_tmp,
        )
        self.update_metadata_value(
            key="dimensions",
            values=dimensions_tmp,
        )
        self.update_metadata_value(
            key="transforms",
            values=transforms_tmp,
        )

        self.database_file_controller.update(
            indices_global={
                k: self.local2global_index_map[k] for k in keys_to_consider
            },
            data=data_tmp,
            msg=f"Updating database for keys: {keys_to_consider}\
            after column pruning",
        )

        self.data_changed()

def remove_duplicate_rows(
    self,
    duplicate_row_eps: float,
) -> None:
    """Iterates over all pairs of rows in this database and detects
    duplicate entries. Then it removes the duplicities from the filesystem.

    Must be called by ALL MPI processes.

    Args:
        duplicate_row_eps (float): Duplicity detection epsilon (|A-B| <=
            epsilon ⇒ duplicate).
    """

    removed_rows = set()
    nentries_global = self.get_n_entries_global()

    # we process the data in batches for less memory intensive algorithm
    # each batch contains the indices of rows in it
    batch_size = nentries_global // (mpi.get_world_size() * 4)
    batches = []
    for idx in range(0, nentries_global, batch_size):
        batches.append((idx, min(nentries_global, idx + batch_size)))
    nbatches = len(batches)

    # then we construct all pairs of batches which need to be compared
    batch_pairs = []
    for b1 in range(nbatches):
        for b2 in range(b1, nbatches):
            batch_pairs.append((b1, b2))

    # each MPI process has a subset of the batch->batch interaction
    mybatch_incidence = [[] for _ in range(nbatches)]
    for u, v in mpi.iterator(batch_pairs):
        mybatch_incidence[u].append(v)

    # now each process finds duplicate entries in its batch->batch subset
    similar_row_pairs = set()
    for u in range(nbatches):
        Vs = mybatch_incidence[u]
        if len(Vs) == 0:
            continue
        indices_global_u = [v for v in range(batches[u][0], batches[u][1])]

        # we load the data corresponding to one batch
        data_u = self.database_file_controller.load_subset(
            indices_global=indices_global_u,
            keys=list(self.get_data_keys()),
            msg=f"  Loading {u}/{nbatches}-th portion of the data",
        )

        U = np.hstack([np.vstack(data_u[k]) for k in data_u.keys()])
        Uindices = np.lexsort(U.T, axis=-1)

        for v in Vs:
            if v == u:
                V = U
                Vindices = Uindices
                indices_global_v = indices_global_u
            else:
                indices_global_v = [
                    i for i in range(batches[v][0], batches[v][1])
                ]

                # or load the data corresponding to the paired batch
                data_v = self.database_file_controller.load_subset(
                    indices_global=indices_global_v,
                    keys=list(self.get_data_keys()),
                    msg=f"    Loading {v}/{nbatches}-th portion of the data",
                )
                V = np.hstack([np.vstack(data_v[k]) for k in data_v.keys()])
                Vindices = np.lexsort(V.T, axis=-1)

                del data_v

            # we detect similar rows in an optimized manner
            similar_row_pairs |= get_similar_row_pairs(
                U,
                Uindices,
                indices_global_u,
                V,
                Vindices,
                indices_global_v,
                duplicate_row_eps,
            )

        del data_u

    # now its time to construct a global set of duplicate rows
    similar_row_pairs_gather = mpi.gather(similar_row_pairs, root=0)
    if mpi.get_rank() == 0:
        similar_row_pairs = set()
        for S in similar_row_pairs_gather:
            similar_row_pairs |= S

        # now, one row can be similar to many other rows, we cant delete
        # them willy-nilly
        # we keep track of how many similarities there are for each row and
        # construct an undirected graph
        cardinality = np.zeros(nentries_global, dtype=int)
        neighborhood = [set() for _ in range(nentries_global)]
        for u, v in similar_row_pairs:
            neighborhood[u].add(v)
            neighborhood[v].add(u)
            cardinality[u] += 1
            cardinality[v] += 1

        # we are interested in rows who have the most similarities
        cardinality_indexing = np.argsort(cardinality)[::-1]

        for u in cardinality_indexing:

            # if the rows was already removed, we cannot remove its
            # neighbours
            if u in removed_rows:
                continue

            # remove all the neghbours of the row
            removed_rows |= neighborhood[u]

    # collected the removed rows for decision purposes
    removed_rows = mpi.broadcast(removed_rows, root=0)
    nremoved = len(removed_rows)

    if nremoved > 0:
        Logger.info(
            f"About to remove {nremoved} rows via row similarity:\
            {list(removed_rows)}"
        )

        removed_indices = list(removed_rows)

        self.database_file_controller.remove(
            keys=list(self.get_data_keys()),
            indices_global=removed_indices,
            msg="Removing entries due to similarity",
        )

        self.data_changed()
        self.add_n_entries(-len(removed_rows))

    @Logger.logged()
def transform_range(
    self,
    keys: list[str],
    target_range: tuple[float, float],
) -> None:
    """For each of the supplied database prefix keys, we perform linear
    scaling and shift operations so the resulting values lie in the
    specified range.

    Args:
        keys (list[str]): A list of database prefix keys for which to\
        perform the transformation.
        target_range (tuple): Target minimal and maximal values of the\
        transformed data ranges.
    """
    min_values = self.get_statistic(stat_code=StatCode.MIN, data_keys=keys)
    max_values = self.get_statistic(stat_code=StatCode.MAX, data_keys=keys)
    value_lo = target_range[0]
    value_hi = target_range[1]

    for k, v in min_values.items():
        for idx, w in enumerate(v):
            Logger.warning(
                f"{k}: {self.get_label(key = k, idx = idx)} ->\
                {w} - {max_values[k][idx]}"
            )

    transform_a = {}
    transform_b = {}
    for key in keys:
        transform_a[key] = np.ones_like(min_values[key])
        transform_b[key] = np.zeros_like(min_values[key])

        n = len(min_values[key])
        for i in range(n):
            if max_values[key][i] - min_values[key][i] > 1e-12:
                transform_a[key][i] = (value_hi - value_lo) / (
                    max_values[key][i] - min_values[key][i]
                )
                transform_b[key][i] = (
                    value_hi - max_values[key][i] * transform_a[key][i]
                )

            Logger.warning(
                f"{key}: {self.get_label(key = key, idx = i)} ->\
                {transform_a[key][i]}x + {transform_b[key][i]}"
            )
    self.load_database(keys=keys)

    results = self._transform_range(
        data=self.data,
        keys=keys,
        A=transform_a,
        B=transform_b,
    )
    transformed_data = results["data"]

    for k in keys:
        data_tmp = transformed_data[k]
        min_values = np.min(data_tmp, axis=0)
        max_values = np.max(data_tmp, axis=0)
        for idx, w in enumerate(min_values):
            Logger.warning(
                f"{k}: {self.get_label(key = k, idx = idx)} ->\
                {w} - {max_values[idx]}"
            )

    self.database_file_controller.update(
        data=transformed_data,
        indices_global={k: self.local2global_index_map[k] for k in keys},
        msg=f"Updating database after transformation of keys: {keys}",
    )

    self.data_changed()
    del transformed_data

    current_transforms = self.metadata_get_value("transforms")
    current_transforms_tmp = {}
    # (ax+b)*c + d = (a*c)x + (b*c + d)
    for key in keys:
        Ta, Tb = current_transforms[key]
        current_transforms_tmp[key] = (
            Ta * transform_a[key],
            Tb * transform_a[key] + transform_b[key],
        )

    self.update_metadata_value(
        key="transforms",
        values=current_transforms_tmp,
    )
