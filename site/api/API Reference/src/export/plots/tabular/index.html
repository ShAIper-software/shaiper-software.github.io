    @staticmethod
@Logger.logged()
def __calculate_histograms_2d_asymmetric(
    indices1: list[int],
    indices2: list[int],
    bin_edges_list1: list[np.ndarray],
    bin_edges_list2: list[np.ndarray],
    data1: np.ndarray,
    data2: np.ndarray,
):
    n1, n2, nvalues_max_2d = len(indices1), len(indices2), 0

    data_raw = [[None for _ in range(n2)] for _ in range(n1)]

    for i1, j1 in enumerate(indices1):
        for i2, j2 in enumerate(indices2):

            data_local = np.histogram2d(
                x=data1[:, j1],
                y=data2[:, j2],
                bins=[
                    bin_edges_list1[i1],
                    bin_edges_list2[i2],
                ],
            )[0]

            data_global = mpi.reduce(data_local, mpi.OP.SUM, root=0)
            data_raw[i1][i2] = data_global

            nvalues_max_2d = max(nvalues_max_2d, np.max(data_global))

    return data_raw, nvalues_max_2d

    @staticmethod
@Logger.logged()
def __export_box_plot_helper(
    categories: list[str],
    values_min: np.ndarray,
    values_q1: np.ndarray,
    values_median: np.ndarray,
    values_q3: np.ndarray,
    values_max: np.ndarray,
    target_dir: str,
    prefix: str,
    indices: list[int],
):
    # Convert summary statistics to the format required by seaborn
    box_data = []
    ncategories = len(indices)
    # fig_height = max(2, ncategories * 1.5)

    image_filenames = []

    if mpi.get_rank() == 0:
        for i in indices:
            box_data.append(
                {
                    "Category": categories[i],
                    "Min": values_min[i],
                    "Q1": values_q1[i],
                    "Median": values_median[i],
                    "Q3": values_q3[i],
                    "Max": values_max[i],
                }
            )

        for j in mpi.tqdm(
            range(len(indices)), desc=f"Exporting Box Plots for: {prefix}"
        ):
            fig, ax = plt.subplots(figsize=(12, 2))
            i = indices[j]
            box = ax.bxp(
                [
                    {
                        "whislo": values_min[i],  # Minimum
                        "q1": values_q1[i],  # First quartile
                        "med": values_median[i],  # Median
                        "q3": values_q3[i],  # Third quartile
                        "whishi": values_max[i],  # Maximum
                        "fliers": [],  # No outliers as we only have summaries
                    }
                ],
                positions=[j],
                showfliers=False,
                vert=False,
                widths=0.6,
                patch_artist=True,  # Enables custom box face colors
                boxprops=dict(
                    edgecolor="black", linewidth=1.5
                ),  # Box color
                medianprops=dict(
                    color="red", linewidth=2
                ),  # Median line color
                capprops=dict(
                    color="black", linewidth=1.5
                ),  # Whisker cap color
                whiskerprops=dict(
                    color="black", linewidth=1.5
                ),  # Whisker color
            )
            for patch in box["boxes"]:
                patch.set_facecolor("#6495ED")

            # Set category labels
            # ax.set_yticks(range(ncategories))
            # ax.set_yticklabels([categories[i] for i in indices])
            ax.set_title(f"Boxplot of {prefix} '{categories[i]}' values")
            ax.set_xlabel("Values")

            plt.tight_layout()  # to fix labeling cut off
            fn = f"{target_dir}/{prefix}_{i}.png"
            plt.savefig(fn)
            plt.close()
            image_filenames.append(fn)
            # image_filenames.append(os.path.abspath(fn))

    return image_filenames

    @staticmethod
@Logger.logged()
def __export_pair_plot_helper(
    compact,
    bin_edges_list_1,
    bin_edges_list_2,
    weights_list_1,
    weights_list_2,
    labels_1,
    labels_2,
    density_matrix_list,
    indices1,
    indices2,
    max_y_lim_1d,
    max_y_lim_2d,
    bin_centers_list_1,
    bin_centers_list_2,
    target_dir,
    prefix,
):
    if mpi.get_rank() > 0:
        return

    image_filenames = []

    if compact:
        nrows = len(indices2) + 1
        ncols = len(indices1) + 2
        width_ratios = [4] * (ncols - 2) + [1, 1]
        height_ratios = [1] + [4] * (nrows - 1)
        gs = gridspec.GridSpec(
            nrows=nrows,
            ncols=ncols,
            width_ratios=width_ratios,
            height_ratios=height_ratios,
            wspace=0.00,
            hspace=0.00,
        )

        sns.set_theme(style="white")

        fig = plt.figure(figsize=(sum(width_ratios), sum(height_ratios)))
        plt.title("Density Plots")
        plt.axis("off")

    valid_indices = []
    for index1, i1 in enumerate(indices1):
        for index2, i2 in enumerate(indices2):
            density_matrix = density_matrix_list[index1][index2]
            if density_matrix is not None:
                valid_indices.append([(index1, index2), (i1, i2)])

    for I in mpi.tqdm(valid_indices, desc=f"Generating pair plots"):
        index1 = I[0][0]
        index2 = I[0][1]
        i1 = I[1][0]
        i2 = I[1][1]

        density_matrix = density_matrix_list[index1][index2]

        x_edges = bin_edges_list_1[index1]
        y_edges = bin_edges_list_2[index2]
        hist_x = weights_list_1[index1]
        hist_y = weights_list_2[index2]
        labels_x = labels_1[indices1[index1]]
        labels_y = labels_2[indices2[index2]]

        if not compact:
            sns.set_theme(style="white")
            plt.title("Density Plot")
            plt.axis("off")
            include_bottom = True
            include_left = True
            width_ratios = [4, 1, 1]
            height_ratios = [1, 4]
            gs = gridspec.GridSpec(
                nrows=2,
                ncols=2,
                width_ratios=width_ratios,
                height_ratios=height_ratios,
                wspace=0.00,
                hspace=0.00,
            )
            fig = plt.figure(
                figsize=(sum(width_ratios), sum(height_ratios))
            )
            ax_main = plt.subplot(gs[1, 0])
            ax_top = plt.subplot(gs[0, 0], sharex=ax_main)
            ax_right = plt.subplot(gs[1, 1], sharey=ax_main)

        else:
            include_bottom = False
            include_left = False
            ax_top = None
            ax_right = None
            ax_main = plt.subplot(gs[nrows - 1 - index2, index1])
            if index2 == len(indices2) - 1:
                ax_top = plt.subplot(gs[0, index1], sharex=ax_main)
            if index1 == len(indices1) - 1:
                ax_right = plt.subplot(
                    gs[nrows - 1 - index2, -1], sharey=ax_main
                )
            if index2 == 0:
                include_bottom = True
            if index1 == 0:
                include_left = True

        ax_main.imshow(
            density_matrix.T,
            origin="lower",
            aspect="auto",
            extent=[x_edges[0], x_edges[-1], y_edges[0], y_edges[-1]],
            # vmin=0, vmax=max_y_lim_2d,
            cmap="rocket",
        )
        for spine in ax_main.spines.values():
            #     spine.set_visible(False)
            spine.set_edgecolor("cyan")  # Set spine color
            spine.set_linewidth(
                1 * max(nrows, ncols)
            )  # Set spine thickness
            # spine.set_linestyle('--')         # Optional: dashed style

        # Histogram on the top (X-axis)
        if ax_top is not None:
            ax_top.bar(
                (x_edges[1:] + x_edges[:-1]) * 0.5,
                hist_x,
                width=np.diff(x_edges),
                color="black",
                alpha=0.8,
            )
            ax_top.set_xlim(x_edges[0], x_edges[-1])
            ax_top.set_ylim(0, max_y_lim_1d)
            ax_top.tick_params(
                axis="x",
                which="both",
                bottom=False,
                top=False,
                labelbottom=False,
            )
            ax_top.set_yticks([])  # Remove y-ticks for cleanliness
            for spine in ax_top.spines.values():
                spine.set_visible(False)

        if ax_right is not None:
            ax_right.barh(
                (y_edges[1:] + y_edges[:-1]) * 0.5,
                hist_y,
                height=np.diff(y_edges),
                color="black",
                alpha=0.8,
            )
            ax_right.set_ylim(y_edges[0], y_edges[-1])
            ax_right.set_xlim(0, max_y_lim_1d)
            ax_right.tick_params(
                axis="y",
                which="both",
                left=False,
                right=False,
                labelleft=False,
            )
            ax_right.set_xticks([])  # Remove y-ticks for cleanliness
            for spine in ax_right.spines.values():
                spine.set_visible(False)

        if include_bottom:
            ax_main.set_xlabel(labels_x)
            ax_main.set_xticks(
                [float(f"{v:0.2f}") for v in bin_centers_list_1[index1]]
            )  # X-axis ticks at column indices
        else:
            ax_main.set_xticks([])

        if include_left:
            ax_main.set_ylabel(labels_y)
            ax_main.set_yticks(
                [float(f"{v:0.2f}") for v in bin_centers_list_2[index2]]
            )  # X-axis ticks at column indices
        else:
            ax_main.set_yticks([])

        if not compact:
            plt.tight_layout()
            fn = f"{target_dir}/{prefix}_{i1:03d}_{i2:03d}.png"
            plt.savefig(fn)
            image_filenames.append(fn)
            plt.close()

    if compact:
        plt.tight_layout()
        fn = f"{target_dir}/{prefix}.png"
        plt.savefig(fn)
        image_filenames.append(fn)
        plt.close()

    return image_filenames

def calculate_cummulative_distribution(
    *,
    data: np.array,
    nbins: int,
) -> tuple[np.array, np.array]:
    """Takes the input flattened numpy array and calculates a cummulative
    value distribution curve with 'nbins' values at the x-axis. The curve
    represents the answer to a question "What portion of data (y axis) has
    value v (x axis) or lower?"

    Args:
        data (np.array): A flattened numpy array containing the values.
        nbins (int): Resolution of the function

    Returns:
        tuple[np.array, np.array]: x and y values
    """
    if data is None:
        msg = "Data must not be None"
        Logger.warning(msg)
        raise ValueError(msg)

    data = np.asarray(data).ravel()
    data = data[np.isfinite(data)]

    if data.size == 0:
        msg = "Data must contain at least one finite number"
        Logger.warning(msg)
        raise ValueError(msg)

    if nbins <= 1:
        msg = "nbins must be greater than 1"
        Logger.warning(msg)
        raise ValueError(msg)

    data_sorted = np.sort(data)
    data_min, data_max = data_sorted[0], data_sorted[-1]

    x_values = np.linspace(data_min, data_max, nbins, endpoint=True)
    counts = np.searchsorted(data_sorted, x_values, side="right")
    y_values = counts / data_sorted.size

    return x_values, y_values

@Logger.logged()
def calculate_histograms(
    *,
    predefined_min: float = None,
    predefined_max: float = None,
    data: np.ndarray,
    nbins: int,
):
    if predefined_min is not None and predefined_max is not None:
        value_range = (predefined_min, predefined_max)

    if len(data.shape) <= 1:
        if predefined_min is None or predefined_max is None:
            min_local = data.min()
            max_local = data.max()
            min_global = mpi.allreduce(min_local, mpi.OP.MIN)
            max_global = mpi.allreduce(max_local, mpi.OP.MAX)
            value_range = (min_global, max_global)

        counts, bin_edges = np.histogram(
            data, bins=nbins, range=value_range
        )
        counts = mpi.reduce(counts, mpi.OP.SUM, root=0)
        bin_centers = (bin_edges[:-1] + bin_edges[1:]) * 0.5
        # print(counts)
        weights = counts / np.sum(counts)

        return bin_centers, weights

    if predefined_min is None or predefined_max is None:
        min_local = data.min(axis=0)
        max_local = data.max(axis=0)
        min_global = mpi.allreduce(min_local, mpi.OP.MIN)
        max_global = mpi.allreduce(max_local, mpi.OP.MAX)
        value_range = [
            (min_global[i], max_global[i]) for i in range(data.shape[1])
        ]
    else:
        value_range = [(predefined_min, predefined_max)] * data.shape[1]

    bin_centers = np.zeros((data.shape[1], nbins))
    weights = np.zeros((data.shape[1], nbins))
    for i in range(data.shape[1]):
        counts, bin_edges = np.histogram(
            data[:, i], bins=nbins, range=value_range[i]
        )
        counts = mpi.reduce(counts, mpi.OP.SUM, root=0)
        bin_centers[i] = (bin_edges[:-1] + bin_edges[1:]) * 0.5
        weights[i] = counts / sum(counts)

    return bin_centers, weights

    @staticmethod
@Logger.logged()
def export_value_distribution(
    database: DatabaseTabular,
    nbins: int,
    indices: dict[list[int]] = None,
    target_dir: str = "exports/tabular/plots",
):
    """_summary_

    Args:
        database (DatabaseTabular): _description_
        nbins (int): _description_
        input_indices (list[int], optional): _description_. Defaults to None.
        output_indices (list[int], optional): _description_. Defaults to None.
        target_dir (str, optional): _description_. Defaults to "exports/tabular/plots".
    """
    if mpi.is_master():
        if not os.path.exists(target_dir):
            os.makedirs(target_dir)
    mpi.sync()

    Logger.info(f"Calculating data for distribution plot.")

    data_keys = list(database.get_data_keys())
    values_min = database.get_statistic(
        data_keys=data_keys,
        stat_code=ENUM_StatisticsCode_Tabular.MIN,
    )
    values_max = database.get_statistic(
        data_keys=data_keys,
        stat_code=ENUM_StatisticsCode_Tabular.MAX,
    )
    if indices is None:
        indices = {k: None for k in data_keys}

    image_filenames = {k: [] for k in data_keys}
    for key in data_keys:
        dim = database.get_dim(key=key)
        if dim <= 0:
            continue

        data = np.array(database.get_data(key=key))
        if indices[key] is None:
            indices[key] = [i for i in range(dim)]

        image_filenames[key] += Plotting.__export_value_distribution_helper(
            target_dir=target_dir,
            prefix=f"distribution_{key}",
            nbins=nbins,
            data=data,
            indices=indices[key],
            values_min=values_min[key] - 1e-6,
            values_max=values_max[key] + 1e-6,
            data_labels=database.metadata_get_value("labels")[key],
        )

    return image_filenames
