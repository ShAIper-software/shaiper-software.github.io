      const edges = [];
      for(let i=0;i<maxRes;i+=K){{ counts.push(sliderData.finest_counts.slice(i,i+K).reduce((a,b)=>a+b,0)); edges.push(sliderData.finest_bins[i]); }}

    def serialize_dashboard(dashboard_structure):
    def convert(obj):
        if isinstance(obj, dict):
            return {k: convert(v) for k, v in obj.items()}
        elif isinstance(obj, list):
            return [convert(x) for x in obj]
        elif hasattr(obj, "to_plotly_json"):
            return pio.to_json(obj)
        else:
            return obj
    return convert(dashboard_structure)
