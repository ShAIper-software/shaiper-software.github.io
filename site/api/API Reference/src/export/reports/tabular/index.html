def __HTML_add_data_correlations(
    database: DatabaseTabular,
    target_dir: str,
):

    image_filenames = PT.export_correlation_matrix(
        database=database,
        target_dir=f"{target_dir}/correlation",
    )

    if image_filenames["inputs-inputs"] is not None:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{CORRELATION_IMAGE_I2I}",
            image_filenames["inputs-inputs"][len(target_dir) + 1 :],
        )

    if image_filenames["inputs-outputs"] is not None:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{CORRELATION_IMAGE_I2O}",
            image_filenames["inputs-outputs"][len(target_dir) + 1 :],
        )

    if image_filenames["outputs-outputs"] is not None:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{CORRELATION_IMAGE_O2O}",
            image_filenames["outputs-outputs"][len(target_dir) + 1 :],
        )

    @staticmethod
@Logger.logged()
def __HTML_add_dropdown_options(
    database: DatabaseTabular,
    target_dir: str,
):
    labels = database.metadata_get_value("labels")

    image_filenames_distributions = PT.export_value_distribution(
        database=database,
        nbins=50,
        target_dir=f"{target_dir}/distributions",
    )
    image_filenames_boxplots = PT.export_box_plot(
        database=database,
        target_dir=f"{target_dir}/boxplots",
    )

    image_filenames_distributions_inputs = image_filenames_distributions[
        "inputs"
    ]
    image_filenames_boxplots_inputs = image_filenames_boxplots["inputs"]
    if len(image_filenames_distributions_inputs) > 0:
        data_inputs = ""
        for i, label in enumerate(labels['inputs']):
            data_inputs += f'<option value="{image_filenames_distributions_inputs[i][len(target_dir) + 1:]}">Input {label}</option>'

        ReporterTabular.body = ReporterTabular.body.replace(
            "{OPTIONS_INPUTS_DISTRIBUTION}",
            data_inputs,
        )

    if len(image_filenames_boxplots_inputs) > 0:
        data_inputs = ""
        for i, label in enumerate(labels['inputs']):
            data_inputs += f'<option value="{image_filenames_boxplots_inputs[i][len(target_dir) + 1:]}">Input {label}</option>'

        ReporterTabular.body = ReporterTabular.body.replace(
            "{OPTIONS_INPUTS_BOXPLOTS}",
            data_inputs,
        )

    image_filenames_distributions_outputs = image_filenames_distributions[
        "outputs"
    ]
    image_filenames_boxplots_outputs = image_filenames_boxplots["outputs"]
    if len(image_filenames_distributions_outputs) > 0:
        data_outputs = ""
        for i, label in enumerate(labels['outputs']):
            data_outputs += f'<option value="{image_filenames_distributions_outputs[i][len(target_dir) + 1:]}">Output {label}</option>'

        ReporterTabular.body = ReporterTabular.body.replace(
            "{OPTIONS_OUTPUTS_DISTRIBUTION}",
            data_outputs,
        )

    if len(image_filenames_boxplots_outputs) > 0:
        data_outputs = ""
        for i, label in enumerate(labels['outputs']):
            data_outputs += f'<option value="{image_filenames_boxplots_outputs[i][len(target_dir) + 1:]}">Output {label}</option>'

        ReporterTabular.body = ReporterTabular.body.replace(
            "{OPTIONS_OUTPUTS_BOXPLOTS}",
            data_outputs,
        )

    @staticmethod
@Logger.logged()
def __HTML_add_submenu(
    database: DatabaseTabular,
):
    dim_inputs = database.get_dim('inputs')
    dim_outputs = database.get_dim('outputs')

    if dim_inputs > 0:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL2_INPUTS}",
            '<button class="tab-button active" onclick="showSubTab(\'Inputs\')">Inputs</button>',
        )
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL3_INPUTS-INPUTS}",
            '<button class="tab-button active" onclick="showSubTab(\'Inputs to Inputs\')">Inputs to Inputs</button>',
        )
    else:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL2_INPUTS}",
            "",
        )
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL3_INPUTS-INPUTS}",
            "",
        )
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL3_INPUTS-OUTPUTS}",
            "",
        )

    if dim_outputs > 0:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL2_OUTPUTS}",
            '<button class="tab-button" onclick="showSubTab(\'Outputs\')">Outputs</button>',
        )
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL3_OUTPUTS-OUTPUTS}",
            '<button class="tab-button" onclick="showSubTab(\'Outputs to Outputs\')">Outputs to Outputs</button>',
        )
    else:
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL2_OUTPUTS}",
            "",
        )
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL3_OUTPUTS-OUTPUTS}",
            "",
        )
        ReporterTabular.body = ReporterTabular.body.replace(
            "{TABS_LEVEL3_INPUTS-OUTPUTS}",
            "",
        )

    ReporterTabular.body = ReporterTabular.body.replace(
        "{TABS_LEVEL3_INPUTS-OUTPUTS}",
        '<button class="tab-button" onclick="showSubTab(\'Inputs to Outputs\')">Inputs to Outputs</button>',
    )

    ReporterTabular.body = ReporterTabular.body.replace(
        "{TABS_LEVEL4_POLYNOMIAL}",
        '<button class="tab-button" onclick="showSubTab(\'Polynomial\')">Polynomial</button>',
    )
    ReporterTabular.body = ReporterTabular.body.replace(
        "{TABS_LEVEL4_GONIOMETRIC_SINE}",
        '<button class="tab-button" onclick="showSubTab(\'Goniometric-Sine\')">Goniometric-Sine</button>',
    )
    ReporterTabular.body = ReporterTabular.body.replace(
        "{TABS_LEVEL4_GONIOMETRIC_COSINE}",
        '<button class="tab-button" onclick="showSubTab(\'Goniometric-Cosine\')">Goniometric-Cosine</button>',
    )

    @Logger.logged()
    def __add_boxplots(
        database: DatabaseTabular,
        target_dir: str,
        prefix: str,
    ):

        image_filenames = PT.export_box_plot(
            database=database,
            target_dir=target_dir,
        )

        dim_outputs = database.metadata_get_value("dim_outputs")
        labels_outputs = database.metadata_get_value("labels_outputs")
        if dim_outputs > 0 and mpi.is_master():
            target_string = """
## Output Values Box-Plots
"""
            for i, label in enumerate(labels_outputs):
                target_string += f"""
Output **{label}**

![{label}]({prefix}{image_filenames['outputs'][i]})
"""
            ReporterTabular.body = ReporterTabular.body.replace(
                "{BOXPLOTS_OUTPUTS}",
                target_string,
            )
        else:
            ReporterTabular.body = ReporterTabular.body.replace(
                "{BOXPLOTS_OUTPUTS}",
                "",
            )

        dim_inputs = database.metadata_get_value("dim_inputs")
        labels_inputs = database.metadata_get_value("labels_inputs")
        if dim_inputs > 0 and mpi.is_master():
            target_string = """
## Input Values Box-Plots
"""
            for i, label in enumerate(labels_inputs):
                target_string += f"""
Input **{label}**

![{label}]({prefix}{image_filenames['inputs'][i]})
"""
            ReporterTabular.body = ReporterTabular.body.replace(
                "{BOXPLOTS_INPUTS}",
                target_string,
            )
        else:
            ReporterTabular.body = ReporterTabular.body.replace(
                "{BOXPLOTS_INPUTS}",
                "",
            )

    @staticmethod
@Logger.logged()
def export_to_html(
    *,
    database: DatabaseTabular,
    tgt_fn: str,
):
    report_data = {
        'Database/Distribution/Total': None,
        'Database/Distribution/Training': None,
        'Database/Distribution/Testing': None,
        'Database/Distribution/Validation': None,

        'Database/Correlation/Total': None,
        'Database/Correlation/Training': None,
        'Database/Correlation/Testing': None,
        'Database/Correlation/Validation': None,

        'Database/Covariance/Total': None,
        'Database/Covariance/Training': None,
        'Database/Covariance/Testing': None,
        'Database/Covariance/Validation': None,

        'Database/Pairwise Distributions/Total': None,
        'Database/Pairwise Distributions/Training': None,
        'Database/Pairwise Distributions/Testing': None,
        'Database/Pairwise Distributions/Validation': None,
    }

    labels_all = database.metadata_get_value("labels")
    data_keys = sorted(list(database.get_data_keys()))
    dimensions = {k: database.get_dim(k) for k in data_keys}
    nentries = np.sum([dimensions[k] for k in dimensions])

    ################## DATA DISTRIBUTIONS ##################
    nbins = 2*3*5*7*11
    data_distribution_total = {}
    for key in data_keys:
        if dimensions[key] <= 0:
            continue

        data = np.array(database.get_data(key=key))
        labels = sorted(list(labels_all[key]))
        bin_centers, weights  = PT.calculate_histograms(
            data = data,
            nbins = nbins,
        )

        for idx, label in enumerate(labels):
            data_distribution_total[label] = {
                'x': [float(v) for v in bin_centers[idx]],
                'y': [float(w) for w in weights[idx]],
            }

    report_data[f"Database/Distribution/Total"] = {
        'histogram_plots': {
            'xlabel': 'Value',
            'ylabel': 'Occurence [%]',
            'data': data_distribution_total
        },
    }
    ################## DATA CORRELATIONS ##################
    stats_correlation_pearson = database.get_statistic(
        data_keys = data_keys,
        stat_code = ENUM_StatisticsCode_Tabular.CORRELATION_PEARSON,
    )
    data_correlation_total = [[0 for _ in range(nentries)] for _ in range(nentries)]
    labels_correlations = []
    row_index = 0
    for key_rows in data_keys:
        if dimensions[key_rows] <= 0:
            continue
        labels_row = sorted(list(labels_all[key_rows]))
        labels_correlations += labels_row

        col_index = 0
        for key_cols in data_keys:
            if dimensions[key_cols] <= 0:
                continue

            if f'{key_rows}-{key_cols}' in stats_correlation_pearson:
                correlation_subdata = stats_correlation_pearson[f'{key_rows}-{key_cols}']
            elif f'{key_cols}-{key_rows}' in stats_correlation_pearson:
                correlation_subdata = stats_correlation_pearson[f'{key_cols}-{key_rows}'].T
            else:
                msg = f'Unknown data combination key for correraltion retrieval: "{key_rows}", "{key_cols}"'
                Logger.warning(msg)
                raise RuntimeError(msg)

            for i in range(correlation_subdata.shape[0]):
                for j in range(correlation_subdata.shape[1]):
                    data_correlation_total[row_index + i][col_index + j] = float(correlation_subdata[i, j])

            col_index += dimensions[key_cols]

        row_index += dimensions[key_rows]

    report_data[f"Database/Correlation/Total"] = {
        'annotated_matrix_symmetric': {
            'title': 'Pearson\'s Correlation Matrix',
            'xlabel': 'Database labels',
            'ylabel': 'Database labels',
            'labels': labels_correlations,
            'data': data_correlation_total,
            'decimals': 2,
            'colorscale': 'Viridis',
        },
    }

    ################## DATA COVARIANCES ##################
    stats_covariance = database.get_statistic(
        data_keys = data_keys,
        stat_code = ENUM_StatisticsCode_Tabular.COVARIANCE,
    )
    data_covariances_total = [[0 for _ in range(nentries)] for _ in range(nentries)]
    labels_covariances = []
    row_index = 0
    for key_rows in data_keys:
        if dimensions[key_rows] <= 0:
            continue
        labels_row = sorted(list(labels_all[key_rows]))
        labels_covariances += labels_row

        col_index = 0
        for key_cols in data_keys:
            if dimensions[key_cols] <= 0:
                continue

            if f'{key_rows}-{key_cols}' in stats_covariance:
                covariance_subdata = stats_covariance[f'{key_rows}-{key_cols}']
            elif f'{key_cols}-{key_rows}' in stats_covariance:
                covariance_subdata = stats_covariance[f'{key_cols}-{key_rows}'].T
            else:
                msg = f'Unknown data combination key for covariance retrieval: "{key_rows}", "{key_cols}"'
                Logger.warning(msg)
                raise RuntimeError(msg)

            for i in range(covariance_subdata.shape[0]):
                for j in range(covariance_subdata.shape[1]):
                    data_covariances_total[row_index + i][col_index + j] = float(covariance_subdata[i, j])

            col_index += dimensions[key_cols]

        row_index += dimensions[key_rows]

    report_data[f"Database/Covariance/Total"] = {
        'annotated_matrix_symmetric': {
            'title': 'Covariance Matrix',
            'xlabel': 'Database labels',
            'ylabel': 'Database labels',
            'labels': labels_covariances,
            'data': data_covariances_total,
            'decimals': 2,
            'colorscale': 'Viridis',
        },
    }

    ####################### END ######################
    report_dir = tgt_fn.rsplit('/', 1)[0]
    mpi.make_dir(report_dir)
    create_report(fn = tgt_fn, data = [report_data])
