def __init__(
    self,
    fn: str,
    initial_file_size: int = 10**6,
) -> None:
    """Sets all internal state variables to default values. Initializes the
    database file.
    - filename to the database
    - initial size of the database file
    - load queue
    - save queue
    - remove queue
    - move queue

    Args:
        fn (str): Path to the directory storign the related database files.
        initial_file_size (int): Initial size of the database file.\
        Defaults to 10**6.
    """
    self.fn = fn
    self.initial_size = initial_file_size

    self.queue_load = {}
    self.queue_save = {}
    self.queue_remove = {}
    self.queue_move = {}

    if not os.path.exists(self.fn):
        env = lmdb.open(self.fn, map_size=initial_file_size, subdir=True)
        env.close()

    if mpi.is_master():
        if not os.path.isdir(self.fn):
            Logger.info(
                f'Creating directory "{self.fn}" containing\
                the data file'
            )
            os.makedirs(self.fn)
    mpi.sync()

    self._load_metadata()

    @Logger.logged()
def append(
    self,
    *,
    data: dict[str],
    msg: str,
) -> None:
    """Takes the supplied data and inserts them at the end of this database.
    Must be called by ALL mpi processes (even if some processes might have
    empty data).

    Args:
        data (dict[str]): A dictionary containing a prefix as a\
        key and the associated array of objects. The controller will\
        store the objects under the path starting with the prefix\
        key.
        msg (str): An informative message for the user's benefit.
    """
    keys = data.keys()
    if self._queue_size() > 0:
        mpi.abort(
            0,
            f"The queue is not empty, cannot save the files with prefix:\
            {keys}",
        )

    # retrieve the global number of new entries for each prefix
    ncurrent = self._get_ncurrently_stored_files(keys)
    nnew_local = {k: len(data[k]) for k in keys}
    nnew = mpi.allgather(data=nnew_local)

    # calculate the appropriate index shifting so no files are overwritten
    # and no gaps in indexing are present
    shift = {k: 0 for k in keys}
    for rank in range(mpi.get_rank() - 1):
        for k in keys:
            shift[k] += nnew[rank][k]

    indices_global = {}
    for k in keys:
        indices_global[k] = [
            ncurrent[k] + shift[k] + i for i in range(nnew_local[k])
        ]

    # finally construct the local save queues
    self.queue_save = {}
    nops = 0
    for k in keys:
        self.queue_save[k] = [
            (self._get_fn_key(k, indices_global[k][i]), data[k][i])
            for i in range(len(indices_global[k]))
        ]
        nops += len(self.queue_save[k])

    Logger.info(
        f'Appending {nops} entries with keys: {keys} into database:\
        "{self.fn}"'
    )

    self._save_files(
        msg,
        synchronous_access=True,
        synchronize=True,
        lmdb_environment_fn=self.fn,
    )

    self._increase_ncurrently_stored_files(
        values={
            k: np.sum([nnew[i][k] for i in range(len(nnew))]) for k in keys
        },
    )

    @Logger.logged()
def load(
    self,
    *,
    keys: list[str],
    msg: str,
) -> tuple[dict[str : list[Any]], list[int]]:
    """Loads all the files with the specified prefix. Each MPI process holds
    a portion of the files. Must be called by ALL MPI processes.

    Args:
        keys (list[str]): A list of prefixes for which to perform the\
        loading operation
        msg (str): An informative message for the user's benefit.

    Returns:
        tuple[dict[str: list[Any]], list[int]]: A list of loaded objects\
        for each supplied prefix and the associated list of global file\
        indices.
    """
    if self._queue_size() > 0:
        mpi.abort(
            0,
            f"The queue is not empty, cannot load the files with prefixes:\
            {keys}",
        )

    nfiles_total = self._get_ncurrently_stored_files(keys)
    self.queue_load = {}
    self.load_buffer = {}
    indices_global = {}
    for key in keys:
        indices_global[key] = [
            i for i in mpi.iterator(range(nfiles_total[key]))
        ]

        self.queue_load[key] = [
            self._get_fn_key(key, i) for i in indices_global[key]
        ]

        self.load_buffer[key] = [
            None for _ in range(len(indices_global[key]))
        ]

    self._load_files(
        msg,
        synchronous_access=False,
        synchronize=False,
        lmdb_environment_fn=self.fn,
    )

    return self.load_buffer, indices_global

def load_single_process(
    self,
    *,
    keys: list[str],
    msg: str,
    rank: int = 0,
) -> tuple[dict[str : list[Any]], list[int]]:
    """Loads all the files with the specified prefix, can be called by any
    MPI processes.

    Args:
        keys (list[str]): A list of prefixes for which to perform the\
        loading operation
        msg (str): An informative message for the user's benefit.
        rank (int): Rank of the process to load data for. Defaults to 0.

    Returns:
        tuple[dict[str: list[Any]], list[int]]: A list of loaded objects\
        for each supplied prefix and the associated list of global file\
        indices.
    """
    if self._queue_size() > 0:
        mpi.abort(
            0,
            f"The queue is not empty, cannot load the files with prefix:\
            {keys}",
        )

    self.queue_load = {k: [] for k in keys}
    self.load_buffer = {}

    if mpi.get_rank() == rank:
        nfiles_total = self._get_ncurrently_stored_files(keys)
        indices_global = list(np.arange(start=0, stop=nfiles_total, step=1))
        for key in keys:
            self.queue_load[key] = [
                self._get_fn_key(key, i) for i in indices_global
            ]

            self.load_buffer[key] = [
                None for _ in range(len(indices_global))
            ]
    else:
        self.load_buffer = {}
        indices_global = []

    self._load_files(
        msg,
        synchronous_access=False,
        synchronize=False,
        lmdb_environment_fn=self.fn,
    )

    return self.load_buffer, indices_global

    @Logger.logged()
def remove(
    self,
    *,
    keys: list[str],
    indices_global: list[int],
    msg: str,
) -> None:
    """Given the list of prefixes and global file indices for each prefix,
    removes the corresponding files. Automatically recalculates which files
    need to be moved so that the resulting files for each prefix use a
    continuous indexation. The number of file io operations is minimized.

    Must be called by ALL mpi processes and assumes ONLY mpi process with
    rank 0 has the relevant data.

    Args:
        keys (list[str]): A list of prefixes for which to perform the\
        removal operation
        indices_global (list[int]): Indices of files to be removed. One\
        list for all prefixes. All processes have the same list, no need\
        for synchronization.
        msg (str): An informative message for the user's benefit.
    """
    if self._queue_size() > 0:
        mpi.abort(
            0,
            f"The queue is not empty, cannot remove the files with prefix:\
            {keys}",
        )

    nremoved = 0
    nops_move = 0
    nops_remove = 0
    # only the master rank will perform this operation
    if mpi.get_rank() == 0:
        self.queue_move = {k: [] for k in keys}
        self.queue_remove = {k: [] for k in keys}

        # sort the global indices for more robust processing
        removed_indices = np.sort(indices_global)
        nremoved = len(removed_indices)
        ntotal = self._get_ncurrently_stored_files(keys)
        assert np.min(ntotal) == np.max(ntotal)

        # keep the indices of kept files in reverse order for more
        # efficient approach
        kept_indices = np.sort(
            list(
                set(range(ntotal[keys[0]])).difference(set(indices_global))
            )
        )[::-1]

        # essentially a tracker of kept/removed file indices
        test_list = [i for i in range(ntotal[keys[0]])]
        for i in removed_indices:
            test_list[i] = None

        for step_idx, kept_idx in enumerate(kept_indices):

            # once the list of files is reduced by 'nremoved' elements,
            # we are done
            if kept_idx < len(test_list) - nremoved:
                break

            # idx_t < kept_idx, we move a file index
            idx_to = removed_indices[step_idx]
            test_list[idx_to] = kept_idx

            for key in keys:
                self.queue_move[key].append(
                    (
                        self._get_fn_key(key, kept_idx),
                        self._get_fn_key(key, idx_to),
                    )
                )

        # now we are in a state when we moved some of the kept files into
        # removed files with lower indices
        # we need to check whether the remaining files to be removed still
        # need removing
        for idx_remove in range(len(test_list) - nremoved, len(test_list)):

            # the index is supposed to be removed, but is not part of the
            # movement logic
            # place it to the remoive queue
            if test_list[idx_remove] is not None:
                for key in keys:
                    self.queue_remove[key].append(
                        self._get_fn_key(key, idx_remove)
                    )
                test_list[idx_remove] = None

        for key in keys:
            nops_remove += len(self.queue_remove[key])
            nops_move += len(self.queue_move[key])

        for i in range(ntotal[keys[0]] - nremoved):
            if test_list[i] is None:
                mpi.abort(
                    0,
                    "Incorrect reindexing map after file removal\
                    (Encountered a deleted element)",
                )
            if test_list[i] in removed_indices:
                mpi.abort(
                    0,
                    "Incorrect reindexing map after file removal\
                    (index should be removed but remained)",
                )

        interr = False
        for i in range(ntotal[keys[0]] - nremoved, ntotal[keys[0]]):
            if test_list[i] is not None:
                Logger.warning(
                    f"Incorrect reindexing map after file removal\
                    (Encountered a valid element[{test_list[i]}])"
                )
                interr = True
        if interr:
            mpi.abort(0, "invalid move/del combinations")

    self._move_files(
        f"Moving {nops_move} files",
        synchronous_access=False,
        synchronize=True,
        lmdb_environment_fn=self.fn,
    )
    self._remove_files(
        f"Removing {nops_remove} files",
        synchronous_access=False,
        synchronize=True,
        lmdb_environment_fn=self.fn,
    )

    nremoved = mpi.broadcast(nremoved, root=0)
    self._increase_ncurrently_stored_files(
        values={k: -nremoved for k in keys},
    )

    arr: dict[str : list[str]],
) -> dict[str:int]:
    """Determines where to start the processing of the supplemented files. This
    is necessary, because the queue is processed in predetrmined chunks, the
    processing of which can fail and need to be restarted.

    Args:
        arr (dict): A dictionary of (key -> filenames) pairs.

    Returns:
        dict: For each key in arr, contains the lowest index of the filename\
        not yet processed
    """
    first_active_index = {k: len(v) for k, v in arr.items()}

    # trivial approach, implement binary search in the future or something more
    # clever with O(1) complexity
    for k, v in arr.items():
        for i in range(len(v)):
            if v[i] is not None:
                first_active_index[k] = i
                break

    return first_active_index

    def lmdb_controller(func)-> Callable:
    """A decorator function used to wrap parallel io methods. It splits the file
    operation into chunks of size 1000. When the database does not have enough
    storage, it increases it by a predefined factor and restarts the file
    processing from the last unfinished chunk.

    The decorated function requires three arguments:
        lmdb_environment_fn (str): path to an LMDB file
        synchronous_access (bool): if true, processes do not access the files\
        simultaneously, but go in a round-robin fashion. When one process\
        finishes all its work, another one line continues with its work.
        synchronize (bool): if true, processes access the files simultaneously.

    Args:
        func (Callable): A function working with files in parallel.

    Returns:
        Callable: the wrapped function
    """

    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        if "lmdb_environment_fn" not in kwargs:
            mpi.abort(
                0,
                f'The function: "{func.__name__}" requires the following\
                argument: "lmdb_environment_fn"',
            )
        if "synchronous_access" not in kwargs:
            mpi.abort(
                0,
                f'The function: "{func.__name__}" requires the following\
                argument: "synchronous_access"',
            )
        if "synchronize" not in kwargs:
            mpi.abort(
                0,
                f'The function: "{func.__name__}" requires the following\
                argument: "synchronize"',
            )

        env_fn = kwargs["lmdb_environment_fn"]
        synchronous_access = kwargs["synchronous_access"]
        synchronize = kwargs["synchronize"]
        kwargs["commit_interval"] = 1000

        env = lmdb.open(env_fn, subdir=True)

        current_env_size = env.info()["map_size"]
        while True:

            kwargs["env"] = env
            try:
                if synchronous_access:
                    for rank in range(mpi.get_world_size()):
                        if rank == mpi.get_rank():
                            func(*args, **kwargs)
                        res = mpi.allreduce(data=0, op=mpi.OP.MAX)
                        if res > 0.5:
                            break
                elif synchronize:
                    func(*args, **kwargs)
                    res = mpi.allreduce(data=0, op=mpi.OP.MAX)
                else:
                    func(*args, **kwargs)
                    res = 0

            except lmdb.MapFullError:
                if synchronous_access or synchronize:
                    res = mpi.allreduce(data=1, op=mpi.OP.MAX)
                else:
                    res = 1

            if res > 0.5:
                new_env_size = int(1.5 * current_env_size)
                Logger.info(
                    f"Setting a new map size:\
                    {_convert_bytes_to_str(current_env_size)} ->\
                    {_convert_bytes_to_str(new_env_size)}"
                )
                current_env_size = new_env_size
                env.set_mapsize(new_env_size)
            else:
                break

        env.close()

    return wrapper
