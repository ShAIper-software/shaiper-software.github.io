    def __init__(
    self,
    *,
    object_uid: str = None,
    f: Any,
    parameters: dict[str, Any] = {},
    action_type: str,
    lazy: bool,
):
    """Depending on which inputs are None, determines what action should be 
    performed.

    - Object initialization and a member function call?
    - A function call?

    The supplied parameters are used to override the default parameters of
    the object construction/function call.

    Args:
        object_uid (str, optional): An object with the function\
            f as an attribute. Defaults to None.
        f (Any): The function to be called, if object_uid refers to a\
            configured object, calls the function as the member of the\
            object class.
        parameters (dict[str, Any], optional): Parameters to be passed into\
            the function call, this overrides the default parameters\
            supplied elsewhere in the configuration hierarchy. Defaults to\
            {}.
        action_type (str): the type of action to be defined, possible\
            values:
            - 'invalid'
            - 'class.static_function(**parameters)'
            - 'global_function(**parameters)'
            - 'member_function(object, **parameters)'
        lazy (bool): A flag determining whether the associated objects\
            should be initialized automatically before (False) the function\
            call or explicitly (True) by the called function.
    """

    if object_uid is None and f is None:
        msg = (
            f"An Action cannot be empty"
            "(requires either an object or a function to be called)"
        )
        Logger.warning(msg)
        raise ValueError(msg)

    self.object_uid = object_uid
    self.object_class = runtime_state.get_object_configuration(
        object_uid=object_uid
    )
    self.f = f
    self.parameters = parameters
    self.action_type = action_type
    self.lazy = lazy

    # check for parameter correctness
    sig = inspect.signature(self.f)
    try:
        if self.action_type == "member_function(object, **parameters)":
            bound = sig.bind(self.object_class, **self.parameters)
            bound.apply_defaults()  # fills in defaults for missing args
        elif self.action_type == "class.static_function(**parameters)":
            bound = sig.bind(**self.parameters)
            bound.apply_defaults()  # fills in defaults for missing args
        elif self.action_type == "global_function(**parameters)":
            bound = sig.bind(**self.parameters)
            bound.apply_defaults()  # fills in defaults for missing args
        else:
            msg = f"Unknown action type: {self.action_type}"
            Logger.warning(msg)
            raise AttributeError(msg)

    except TypeError as e:
        keys_in_SmP, keys_in_PmS = _filter_parameters_extra(
            sig=sig, parameters=parameters
        )
        msg = (
            f'Invalid "{self.f}" init parameters "{list(self.parameters.keys())}" ->'
            f'"{e}"\nPossible candidates:{list(keys_in_SmP)}, '
            f"extra arguments: {list(keys_in_PmS)}"
        )
        Logger.warning(msg)
        raise TypeError(msg)

    def call(
    self,
    *,
    parameters: dict[str, Any] = {},
    reinit: bool,
) -> Any:
    """Calls the function. The supplied parameters override the member
    parameters.

    Args:
        parameters (dict[str, Any]): Explicit override of the function call\
        parameters to be used instead of the default ones.

    Returns:
        Any: An instance of the class.
    """
    active_parameters = _sample_parameters(
        default_parameters=self.parameters,
        override_parameters=parameters,
        lazy=self.lazy,
        reinit=reinit,
    )
    return self._call(parameters=active_parameters, reinit=reinit)

    def __init__(
    self,
    *,
    obj: Any,
    parameters: dict[str, Any],
    object_uid: str,
):
    """
    Stores the object and its parametrization for future use. Processes the
    parameters for efficient future use.
    Args:
        obj (Any): A class representing the object to be initialized.
        parameters (dict[str, Any]): Object parametrization.
    """
    obj = inspect.unwrap(obj)
    self.obj = obj
    self.parameters = parameters
    self.object_uid = object_uid

    # check parameter validity
    sig = inspect.signature(obj.__init__)
    try:
        # remove 'self' parameter from the signature binding
        bound = sig.bind_partial(None, **parameters)
        bound.apply_defaults()

        # drop 'self' from reported arguments
        args = dict(bound.arguments)
        args.pop("self", None)

        # print(object_uid, obj, sig.parameters)
        if "object_uid" in sig.parameters:
            # print(object_uid)
            self.parameters["object_uid"] = object_uid
        # else:
        #     print(f'{self.object_uid} -> NO')

    except TypeError as e:
        keys_in_SmP, keys_in_PmS = _filter_parameters_extra(
            sig=sig, parameters=parameters
        )
        msg = (
            f'Invalid "{self.obj}" init parameters "{list(self.parameters.keys())}" ->'
            f'"{e}"\nPossible candidates:{list(keys_in_SmP)}, '
            f"extra arguments: {list(keys_in_PmS)}"
        )
        Logger.warning(msg)
        raise TypeError(msg)

    runtime_state ¶

    Contains the following static objects which are referenced during the program execution:

    - object_configurations: holds the parameters used in constructors of        all objects.
- objects: holds all the instantiated objects.
- parametrized_object_uids: a list of unique identifiers of objects            requiring randomg parameter sampling
- actions: for each object, holds a sequence of actions to be performed            with this object

    Furthermore, contains the global configuration object related to this runtime.

def add_action(
    *,
    object_uid: str,
    fname: str,
    parameters: dict,
):
    """Adds an action associated with the specified object. The action is a
    class member function to be called with the specified parameters.

    Args:
        object_uid (str): A unique identifier specifying the object.
        fname (str): A name of the function to be called.
        parameters (dict): Parameters to override the default function\
            parameters to the member function.
    """

    try:
        obj = runtime_state.get_object_configuration(object_uid)
        action_type = "invalid"
        if obj is None:
            try:
                action_type = "global_function(**parameters)"
                # test for a global function
                module_path, function_name = fname.rsplit(".", 1)

                module = import_module(module_path)
                f = getattr(module, function_name)
                is_lazy = is_object_lazy(f)

                Logger.info(f"Adding action:")
                Logger.info(f"  lazy: {is_lazy}")
                Logger.info(f"  '{function_name}' from: '{module_path}'")
                Logger.info(f"  action parameters: {parameters}")

            except ModuleNotFoundError as e:
                action_type = "class.static_function(**parameters)"
                # split into module and a function
                # module.Class.function
                module_path, obj_name, function_name = fname.rsplit(
                    ".", 2
                )  # split into module and a function
                module = import_module(module_path)
                obj = getattr(module, obj_name)
                f = getattr(obj, function_name)
                is_lazy = is_object_lazy(f)

                Logger.info(f"Adding action:")
                Logger.info(f"  lazy: {is_lazy}")
                Logger.info(
                    f"  {function_name} from: " f"{module_path}.{obj_name}"
                )
                Logger.info(f"  action parameters: {parameters}")

        else:
            action_type = "member_function(object, **parameters)"
            obj = obj.get_object_class()
            f = getattr(obj, fname)
            is_lazy = is_object_lazy(f)

            Logger.info(f"Adding action:")
            Logger.info(f"  lazy: {is_lazy}")
            Logger.info(f"  object key: {object_uid}")
            Logger.info(f"  object class: {obj}")
            Logger.info(f"  action function: {fname}")
            Logger.info(f"  action parameters: {parameters}")

        Logger.info(f'  Internal action type: "{action_type}"')

        runtime_state.actions.append(
            Action(
                object_uid=object_uid,
                f=f,
                parameters=parameters,
                action_type=action_type,
                lazy=is_lazy,
            )
        )

    except AttributeError as e:
        Logger.warning(e)
        raise AttributeError(e)

def add_object_configuration(
    *,
    key: str,
    constructor: str,
    parameters: dict,
):
    """Given the object unique identifier, stores the class and parameters\
        required for initialization.

    Args:
        key (str): A unique identifier of the object.
        constructor (str): Name of the class representing the object or a\
            function returing an instance of an object.
        parameters (dict): Parameters for initialization.

    Raises:
        ValueError: If the object is already stored, raises this error.
    """


    if runtime_state.contains_uid(key):
        msg = (
            f'An object or a function with ID "{key}" is already'
            "configured. Skipping further configuration..."
        )
        Logger.warning(msg)
        return

    f = None
    try:
        # test for class definition (package.class)
        module_path, obj_name = constructor.rsplit(".", 1)

        module = import_module(module_path)
        f = getattr(module, function_name)

        Logger.info(f"Adding object definition:")
        Logger.info(f"  object UID: '{key}'")
        Logger.info(f"  object class: '{obj_name}' from '{module_path}'")
        Logger.info(f"  object parameters: {parameters}")
    except Exception as e:
        Logger.warning(f'\nEncountered an exception during object definition when treating object as a class: {e}\n  key: {key}\n  constructor: {constructor}\n  parameters: {parameters}\n')
        try:
            # test for global function (package.function)
            module_path, function_name = constructor.rsplit(".", 1)

            module = import_module(module_path)
            f = getattr(module, function_name)

            Logger.info(f"Adding object definition:")
            Logger.info(f"  object UID: '{key}'")
            Logger.info(
                f"  object function: '{function_name}' from '{module_path}'"
            )
            Logger.info(f"  object parameters: {parameters}")
        except Exception as e:
            Logger.warning(f'\nEncountered an exception during object definition when treating object as a global function: {e}\n  key: {key}\n  constructor: {constructor}\n  parameters: {parameters}\n')
            try:
                # test for static function definition (package.class.static_function)
                module_path, obj_name, f_name = constructor.rsplit(".", 2)

                module = import_module(module_path)
                obj = getattr(module, obj_name)
                f = getattr(obj, f_name)

                Logger.info(f"Adding object definition:")
                Logger.info(f"  object UID: '{key}'")
                Logger.info(
                    f"  object class: '{obj_name}' from '{module_path}'"
                )
                Logger.info(
                    f"  function name: '{f_name}' from '{obj_name}'"
                )
                Logger.info(f"  object parameters: {parameters}")
            except Exception as e:
                Logger.warning(f'\nEncountered an exception during object definition when treating object as a static function: {e}\n  key: {key}\n  constructor: {constructor}\n  parameters: {parameters}\n')

    if f is None:
        msg = f"Unable to initialize an object defined at: {constructor} with parameters: {parameters}"
        Logger.warning(msg)
        raise RuntimeError(msg)

    obj_parametrization = ObjectParametrization(
        obj=f,
        parameters=parameters,
        object_uid=key,
    )
    runtime_state.object_configurations[key] = obj_parametrization

    if obj_parametrization.is_parametrized():
        runtime_state.parametrized_object_uids.append(key)

def check_object_dependencies():
    """Analyzes the configured objects and actions and builds a hierarchy
    graph and ordering in which the objects should be initialized and
    actions called.
    """

    Logger.info("Checking object and action dependencies.")

    object_dependencies = {}
    action_dependencies = {}
    for operation in runtime_state.operations:
        operation_object_UIDs = operation["objects"]
        operation_actions = operation["actions"]

        for action_index in operation_actions:
            action = runtime_state._get_action(action_index)
            # Logger.warning(f'{action_index}: {action}')

            action_dependencies[action_index] = []
            if action.object_uid is not None:
                action_dependencies[action_index].append(action.object_uid)

            for (
                parent_object_uid
            ) in runtime_state.get_objects_from_parameters(
                action.parameters
            ):
                action_dependencies[action_index].append(parent_object_uid)
            Logger.info(
                f"[{action_index}] Action Dependencies: {action_dependencies[action_index]}"
            )

        for operation_object_UID in operation_object_UIDs:
            object_dependencies[operation_object_UID] = []
            obj = runtime_state.get_object_configuration(
                operation_object_UID
            )
            for (
                parent_object_uid
            ) in runtime_state.get_objects_from_parameters(obj.parameters):
                object_dependencies[operation_object_UID].append(
                    parent_object_uid
                )
            Logger.info(
                f"[{operation_object_UID}] Object Dependencies: {object_dependencies[operation_object_UID]}"
            )

    cyclic_dependencies = _get_cyclic_dependencies(object_dependencies)
    if len(cyclic_dependencies) > 0:
        msg = (
            "The configuration files contain cyclic dependencies in"
            "object construction..."
        )
        Logger.warning(msg)
        for obj_uid, dep in cyclic_dependencies.items():
            Logger.warning(f"Object: {obj_uid} -> {dep}")

        raise RuntimeError(f"{msg}:  {cyclic_dependencies}")

    runtime_state.objects[object_uid] = obj
